Input Parsing — Text → Structured Intent

The parser must handle complex, multi-constraint sentences and extract zone types, size, priority keywords, and flags like eco-friendly or high-density. Two options: GPT prompt or keyword rules.

Two example inputs this system must handle:
1. "Design a 10 acre eco friendly city with parks, hospitals, low traffic and commercial zones"
2. "Build a smart urban area with healthcare, green spaces, high density housing and balanced traffic"
Keyword → Zone Mapping Dictionary
park / green / nature / garden
→ zone: "park"
hospital / healthcare / clinic / medical
→ zone: "hospital"
residential / housing / homes / living
→ zone: "residential"
commercial / shops / market / business
→ zone: "commercial"
school / education / campus / college
→ zone: "school"
green spaces / eco / sustainable
→ flag: eco:true
low traffic / less traffic
→ traffic: "low"
balanced traffic / medium traffic
→ traffic: "balanced"
high density / dense / compact
→ density: "high"
low density / spacious / open
→ density: "low"
10 acre / 5 acre / [N] acre
→ area_in_acres: N
smart / intelligent / modern
→ flag: smart:true
JavaScript — Keyword Parser
function parseInput(text) {
  const lower = text.toLowerCase();
  const zones = [];

  // Zone keyword maps
  const zoneMap = {
    park: ["park", "green space", "garden", "nature", "green"],
    hospital: ["hospital", "healthcare", "clinic", "medical"],
    residential: ["residential", "housing", "homes", "living", "houses"],
    commercial: ["commercial", "shops", "market", "business"],
    school: ["school", "education", "campus", "college"]
  };

  // Match zones
  for (const [type, keywords] of Object.entries(zoneMap)) {
    if (keywords.some(k => lower.includes(k))) {
      zones.push(type);
    }
  }

  // Always include residential as fallback
  if (!zones.includes("residential")) zones.unshift("residential");

  // Extract area (e.g., "10 acre")
  const areaMatch = lower.match(/(\d+)\s*acre/);
  const area = areaMatch ? parseInt(areaMatch[1]) : 5;

  // Traffic priority
  let traffic = "balanced";
  if (/low traffic|less traffic|minimal traffic/.test(lower)) traffic = "low";
  if (/high traffic|busy|heavy traffic/.test(lower)) traffic = "high";

  // Density
  let density = "medium";
  if (/high density|dense|compact/.test(lower)) density = "high";
  if (/low density|spacious|open/.test(lower)) density = "low";

  // Eco flag
  const eco = /eco|sustainable|green|environment/.test(lower);

  return { zones, area, traffic, density, eco };
}

Grid Engine — Building the Layout Array

The grid is a 2D array of N×N cells. Each cell has a type (zone type or road), a zone ID, and metadata. The engine fills the grid based on the parsed JSON and placement rules.

JavaScript — Grid Initialization + Zone Placement
function buildGrid(config) {
  const N = config.grid_size;
  const grid = Array.from({length: N}, () =>
    Array.from({length: N}, () => ({
      type: "empty", zoneId: null, explanation: ""
    }))
  );

  // Step 1: Place roads first (perimeter + main cross roads)
  placeRoads(grid, N, config.constraints.traffic);

  // Step 2: Place zones in priority order
  const order = ["residential", "park", "hospital", "commercial", "school"];
  for (const zType of order) {
    const zoneCfg = config.zones.find(z => z.type === zType);
    if (!zoneCfg) continue;
    placeZone(grid, N, zoneCfg, config);
  }

  // Step 3: Fill remaining empty as residential
  fillEmpty(grid, N);

  return grid;
}

function placeZone(grid, N, zoneCfg, config) {
  let placed = 0;
  let attempts = 0;

  while (placed < zoneCfg.count && attempts < 1000) {
    attempts++;
    let r, c;

    if (zoneCfg.near_road) {
      // Pick random road cell, try adjacent empty cell
      [r, c] = getAdjacentToRoad(grid, N);
    } else if (zoneCfg.type === "park") {
      // Parks go near residential blocks
      [r, c] = getAdjacentToZone(grid, N, "residential");
    } else {
      // Residential: random empty interior cell
      r = 1 + Math.floor(Math.random() * (N - 2));
      c = 1 + Math.floor(Math.random() * (N - 2));
    }

    if (r !== undefined && grid[r][c].type === "empty") {
      grid[r][c].type = zoneCfg.type;
      grid[r][c].explanation = getExplanation(zoneCfg.type, r, c, grid, N);
      placed++;
    }
  }
}

Road Generation Logic

Roads must connect all zones. The system uses a simple cross (+) pattern based on traffic priority setting from user input.

JavaScript — Road Placement
function placeRoads(grid, N, trafficLevel) {
  const mid = Math.floor(N / 2);

  // Always: perimeter border roads
  for (let i = 0; i < N; i++) {
    grid[0][i].type = "road";
    grid[N-1][i].type = "road";
    grid[i][0].type = "road";
    grid[i][N-1].type = "road";
  }

  // Always: one center cross road
  for (let i = 0; i < N; i++) {
    grid[mid][i].type = "road";
    grid[i][mid].type = "road";
  }

  // Balanced traffic: add 2 more inner roads (grid pattern)
  if (trafficLevel === "balanced") {
    const q1 = Math.floor(N / 4);
    const q3 = Math.floor(N * 3 / 4);
    for (let i = 0; i < N; i++) {
      grid[q1][i].type = "road";
      grid[q3][i].type = "road";
    }
  }

  // High traffic: full grid roads every 4 cells
  if (trafficLevel === "high") {
    for (let i = 0; i < N; i += 4) {
      for (let j = 0; j < N; j++) {
        grid[i][j].type = "road";
        grid[j][i].type = "road";
      }
    }
  }

  // Low traffic: only perimeter + center cross (already done)
}
6
Urban Score Formulas

Score every generated layout on 4 metrics. Formulas are intentionally simple so they are explainable to judges.

Sustainability
score = (park_cells / total_cells) × 100  → normalize to 0–100
Traffic Score
score = 100 − (road_cells / total_cells) × 100  → fewer roads = better for "low traffic"
Walkability
score = 100 − (avg_distance_residential_to_park × 3)  → shorter distance = higher walkability
Density Score
score = (residential_cells / non_road_cells) × 100  → reflects city density
JavaScript — Score Calculator
function calculateScores(grid, N) {
  let park=0, road=0, residential=0, total=N*N;
  const resCells=[], parkCells=[];

  for(let r=0;rfor(let c=0;cconst t=grid[r][c].type;
    if(t==="park"){park++;parkCells.push([r,c]);}
    if(t==="road") road++;
    if(t==="residential"){residential++;resCells.push([r,c]);}
  }

  // Walkability: avg distance from each residential to nearest park
  let totalDist=0;
  resCells.forEach(([r,c])=>{
    const d=parkCells.reduce((min,[pr,pc])=>{
      const dist=Math.abs(r-pr)+Math.abs(c-pc);
      return Math.min(min,dist);
    },999);
    totalDist+=d;
  });
  const avgDist=resCells.length>0?totalDist/resCells.length:0;

  return {
    sustainability: Math.round((park/total)*100 * 3.5),        // scale up for visibility
    traffic: Math.round(100-(road/total)*200),
    walkability: Math.round(Math.max(0, 100-(avgDist*5))),
    density: Math.round((residential/(total-road))*100)
  };
}

Explainability Engine

When user clicks a block, inspect its neighbors and generate a context-aware explanation. This makes the system look intelligent without any AI.

JavaScript — Dynamic Explanation Generator
function getExplanation(type, r, c, grid, N) {
  const neighbors = getNeighborTypes(grid, r, c, N);

  const templates = {
    park: (nb) => {
      if (nb.includes("residential"))
        return "Park placed near residential zone for easy walkability and daily access.";
      return "Park added to improve green coverage and sustainability score.";
    },
    hospital: (nb) => {
      if (nb.includes("road"))
        return "Hospital placed adjacent to main road for quick emergency vehicle access.";
      return "Hospital positioned near road network to ensure accessibility.";
    },
    residential: (nb) => {
      if (nb.includes("park"))
        return "Residential block benefits from proximity to park — improves livability score.";
      if (nb.includes("road"))
        return "Residential zone placed near road for transportation access.";
      return "Residential zone placed in quiet interior — low noise, high livability.";
    },
    commercial: (nb) => {
      if (nb.includes("road"))
        return "Commercial zone on road edge — maximizes customer footfall and access.";
      return "Commercial area placed for economic activity near transport routes.";
    },
    road: () => "Road cell — connects all zones and ensures full layout connectivity.",
    school: (nb) => {
      if (nb.includes("residential"))
        return "School near residential area — reduces commute time for students.";
      return "School placed in safe, low-traffic interior zone.";
    }
  };

  const fn = templates[type];
  return fn ? fn(neighbors) : "Zone placed based on layout optimization rules.";
}

function getNeighborTypes(grid, r, c, N) {
  const dirs = [[-1,0],[1,0],[0,-1],[0,1]];
  return dirs
    .filter(([dr,dc]) => r+dr>=0 && r+dr=0 && c+dcmap(([dr,dc]) => grid[r+dr][c+dc].type);
}

JavaScript — Three.js Grid Renderer
function render3D(grid, N, scene) {
  const COLORS = {
    park: 0x4caf50, residential: 0x2196f3, commercial: 0xff9800,
    hospital: 0xf44336, school: 0x9c27b0, road: 0xe0e0e0, empty: 0xcccccc
  };
  const HEIGHTS = {
    park: 0.3, residential: 1.5, commercial: 2.5,
    hospital: 2.0, school: 1.8, road: 0.1, empty: 0.2
  };
  const CELL = 1.1; // gap between cells

  for (let r = 0; r < N; r++) {
    for (let c = 0; c < N; c++) {
      const t = grid[r][c].type;
      const h = HEIGHTS[t] || 0.2;
      const geo = new THREE.BoxGeometry(1, h, 1);
      const mat = new THREE.MeshLambertMaterial({ color: COLORS[t] || 0xcccccc });
      const cube = new THREE.Mesh(geo, mat);
      cube.position.set(c * CELL, h / 2, r * CELL);
      cube.userData = { type: t, explanation: grid[r][c].explanation };
      scene.add(cube);
    }
  }
}

Fallback Templates

If the AI call fails or the input has no recognizable keywords, fall back to these hardcoded JSON configs. Always show something useful.

JavaScript — Fallback Templates
const FALLBACK_TEMPLATES = {
  eco: {
    zones: [
      {type:"residential",count:10,density:"medium"},
      {type:"park",count:5,priority:"high"},
      {type:"hospital",count:1,near_road:true}
    ],
    flags:{eco:true}, constraints:{traffic:"low",density:"medium"},
    area_in_acres:5, grid_size:20
  },
  urban: {
    zones: [
      {type:"residential",count:14,density:"high"},
      {type:"commercial",count:4,near_road:true},
      {type:"hospital",count:2,near_road:true},
      {type:"park",count:3}
    ],
    flags:{smart:true}, constraints:{traffic:"balanced",density:"high"},
    area_in_acres:10, grid_size:25
  },
  default: {
    zones: [
      {type:"residential",count:8},
      {type:"park",count:2},
      {type:"hospital",count:1,near_road:true}
    ],
    flags:{}, constraints:{traffic:"balanced",density:"medium"},
    area_in_acres:5, grid_size:20
  }
};

function getFallback(text) {
  const lower = text.toLowerCase();
  if (/eco|green|sustainable/.test(lower)) return FALLBACK_TEMPLATES.eco;
  if (/urban|smart|dense|city/.test(lower)) return FALLBACK_TEMPLATES.urban;
  return FALLBACK_TEMPLATES.default;
}

Hackathon Priority: Don't over-engineer edge case handling. Use a simple try/catch + fallback template pattern and move on. A working demo with a fallback beats a broken demo with perfect logic.
Pro Tip for Demo: Before the demo, test these 3 inputs specifically and confirm they produce valid grids: (1) "eco friendly city with parks", (2) "smart urban area with healthcare and high density housing", (3) empty input. Cover all three and your demo is safe.
